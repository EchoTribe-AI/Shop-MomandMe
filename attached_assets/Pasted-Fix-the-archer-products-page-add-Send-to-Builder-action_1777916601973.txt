Add this JavaScript function in the <script> block, after the clearTray() function:
function sendToBuilder(destination) {
  if (!state.tray.length) {
    showToast('Select at least one product first');
    return;
  }
  // Save selected products to sessionStorage for the destination page to read
  const queue = state.tray.map(t => ({
    asin: t.asin,
    name: t.name,
    commission: t.commission,
    link: t.link || state.linkedAsins[t.asin] || null,
  }));
  sessionStorage.setItem('archer_build_queue', JSON.stringify(queue));
  window.location.href = destination;
}
